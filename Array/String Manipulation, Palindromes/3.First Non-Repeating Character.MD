Return the **first character** in the string `s` that **does not repeat**.

## âœ… Code:

```cpp
class Solution {
  public:
    char nonRepeatingChar(string &s) {
        vector<int> freq(26, 0);  // For lowercase 'a' to 'z'
        
        for (char ch : s) {
            freq[ch - 'a']++;     // Count frequency
        }

        for (char ch : s) {
            if (freq[ch - 'a'] == 1) {
                return ch;        // First non-repeating char
            }
        }
       
        return '$';               // Not found
    }
};
```
